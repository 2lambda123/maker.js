I've just wrapped up some documentation on Basic Drawing. There are many simple walkthroughs, code examples, and hands-on demos using JsFiddle. Please take a look, I guarantee it will be worth your time - at the end, you'll know how to create your own skateboard deck!

{% highlight javascript %} var makerjs = require('makerjs'); function truckBolts(x, y) { var tx = 1 + 5/8; var ty = 1 + 1/8; var bolts = new makerjs.models.BoltRectangle('truck', tx, ty, 7/32); bolts.origin = [x + tx / -2, y + ty / -2]; return bolts; } function deck(width, length, truckOffset) { this.id = 'deck'; this.models = [ new makerjs.models.Oval('board', length, width), new truckBolts(truckOffset, width / 2), new truckBolts(length - truckOffset, width / 2) ] } var svg = makerjs.exporter.toSVG(new deck(8, 32, 7)); document.write(svg); {% endhighlight %} try it in JsFiddle ⇗
