--- title: Basic models ---

Maker.js provides these basic models:

  • makerjs.models.BoltCircle
  • makerjs.models.BoltRectangle
  • makerjs.models.ConnectTheDots
  • makerjs.models.GoldenRectangle
  • makerjs.models.Oval
  • makerjs.models.OvalArc
  • makerjs.models.Polygon
  • makerjs.models.Ring
  • makerjs.models.Rectangle
  • makerjs.models.RoundRectangle
  • makerjs.models.SCurve
  • makerjs.models.Square

Here are a few examples to illustrate how you might use these:

House:

{% highlight javascript %} var makerjs = require('makerjs'); var points = [ [100, 0], [100, 100], [0, 175], [-100, 100], [-100, 0], [-20, 0], [-20, 80], [20, 80], [20, 0] ]; var house = new makerjs.models.ConnectTheDots('house', true, points); var window1 = new makerjs.models.Square('window1', 40); window1.origin = [40, 40]; var window2 = new makerjs.models.Square('window2', 40); window2.origin = [-80, 40]; var houseWithWindows = { id: 'myHouse', models: [ house, window1, window2 ] } var svg = makerjs.exporter.toSVG(houseWithWindows); document.write(svg); {% endhighlight %} try it in JsFiddle ⇗

Tablet mount:

{% highlight javascript %} var makerjs = require('makerjs'); var outer = new makerjs.models.RoundRectangle('outer', 200, 280, 8); var inner = new makerjs.models.RoundRectangle('inner', 160, 230, 8); inner.origin = [20, 30]; var buttonhole = new makerjs.paths.Circle('v', [100, 15], 8); var bolts = new makerjs.models.BoltRectangle('bolts', 180, 260, 2); bolts.origin = [10, 10]; var tabletFaceMount = { id: 'tabletFaceMount', paths: [ buttonhole ], models: [ inner, outer, bolts ] }; var svg = makerjs.exporter.toSVG(tabletFaceMount); document.write(svg); {% endhighlight %} try it in JsFiddle ⇗

Circular adapter plate:

{% highlight javascript %} var makerjs = require('makerjs'); var model = { id: 'adapter', models: [ new makerjs.models.Ring('ring1', 40, 100), new makerjs.models.BoltCircle('bc1', 90, 4, 10), new makerjs.models.BoltCircle('bc2', 55, 7, 6, 30) ] }; var svg = makerjs.exporter.toSVG(model); document.write(svg); {% endhighlight %} try it in JsFiddle ⇗

Skateboard deck:

{% highlight javascript %} var makerjs = require('makerjs'); function truckBolts() { var tx = 1 + 5/8; var ty = 1 + 1/8; var bolts = new makerjs.models.BoltRectangle('truck', tx, ty, 7/32 / 2); bolts.origin = [tx / -2, ty / -2]; this.models = [bolts]; } function deck(width, length, truckOffset) { this.id = 'deck'; var board = new makerjs.models.Oval('board', length, width); board.origin = [0, width / -2]; var truck1 = new truckBolts(); truck1.origin = [truckOffset, 0]; var truck2 = new truckBolts(); truck2.origin = [length - truckOffset, 0]; this.models = [board, truck1, truck2]; } var svg = makerjs.exporter.toSVG(new deck(8, 32, 7)); document.write(svg); {% endhighlight %} try it in JsFiddle ⇗
