--- title: Rendering SVG ---

Call the makerjs.exporter.toSVG function and pass your path as a parameter:

{% highlight javascript %} var makerjs = require('makerjs'); var line = { id: "myLineId", type: "line", origin: [0, 0], end: [50, 50] }; var svg = makerjs.exporter.toSVG(line); document.write(svg); {% endhighlight %}

You may also call makerjs.exporter.toSVG with an array of paths as a parameter:

{% highlight javascript %} var makerjs = require('makerjs'); var line = { id: "myLineId", type: "line", origin: [0, 0], end: [50, 50] }; var circle = { id: "myCircleId", type: "circle", origin: [0, 0], radius: 25 }; var pathArray = [ line, circle ]; var svg = makerjs.exporter.toSVG(pathArray); document.write(svg); {% endhighlight %}
