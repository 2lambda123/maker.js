--- title: Rendering SVG ---

Call the makerjs.exporter.toSVG function and pass your path as a parameter:

{% highlight javascript %} var makerjs = require('makerjs'); var line = { id: "myLineId", type: "line", origin: [0, 0], end: [100, 100] }; var svg = makerjs.exporter.toSVG(line); document.write(svg); {% endhighlight %}
